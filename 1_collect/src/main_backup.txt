/*
 * ============================================================
 * xiao_capture.ino
 * Sketch untuk menangkap dataset gambar sampah menggunakan
 * kamera OV2640 pada XIAO ESP32S3 Sense.
 * 
 * Fitur:
 * - Capture 5 frame per sesi (sekuens untuk CNN+LSTM)
 * - Simpan ke SD card dengan format /[kelas]/sesi_[NNN]/f[1-5].jpg
 * - Ganti kelas aktif via Serial Monitor (k/p/o)
 * - Session counter persistent via SPIFFS
 * - Error handling lengkap dengan LED indikator
 * 
 * Hardware: Seeed Studio XIAO ESP32S3 Sense
 * Kamera  : OV2640 (built-in)
 * ============================================================
 */

#include <Arduino.h>
#include "esp_camera.h"
#include "SD.h"
#include "FS.h"
#include "SPIFFS.h"

// ============================================================
// KONFIGURASI PIN KAMERA OV2640 — XIAO ESP32S3 Sense
// Sesuai dokumentasi resmi Seeed Studio
// ============================================================
#define PWDN_GPIO_NUM    -1
#define RESET_GPIO_NUM   -1
#define XCLK_GPIO_NUM    10
#define SIOD_GPIO_NUM    40
#define SIOC_GPIO_NUM    39
#define Y9_GPIO_NUM      48
#define Y8_GPIO_NUM      11
#define Y7_GPIO_NUM      12
#define Y6_GPIO_NUM      14
#define Y5_GPIO_NUM      16
#define Y4_GPIO_NUM      18
#define Y3_GPIO_NUM      17
#define Y2_GPIO_NUM      15
#define VSYNC_GPIO_NUM   38
#define HREF_GPIO_NUM    47
#define PCLK_GPIO_NUM    13

// ============================================================
// KONFIGURASI PIN LAINNYA
// ============================================================
#define SD_CS_PIN        21          // Pin CS untuk SD card
#define BUTTON_PIN       1           // Tombol capture (INPUT_PULLUP)
#define LED_BUILTIN_PIN  21          // LED onboard XIAO ESP32S3 (opsional)

// ============================================================
// KONFIGURASI CAPTURE
// ============================================================
#define FRAMES_PER_SESSION  5        // Jumlah frame per sesi
#define FRAME_DELAY_MS      200      // Jeda antar frame (ms)
#define DEBOUNCE_DELAY_MS   300      // Debounce tombol (ms)

// ============================================================
// NAMA KELAS SAMPAH
// ============================================================
const char* CLASS_NAMES[] = {"kertas", "plastik", "organik"};
const int NUM_CLASSES = 3;

// ============================================================
// VARIABEL GLOBAL
// ============================================================
int kelasAktif = 0;                  // 0=kertas, 1=plastik, 2=organik
int sessionCounters[3] = {0, 0, 0};  // Counter sesi per kelas
bool sdReady = false;                // Status SD card
bool cameraReady = false;            // Status kamera
unsigned long lastButtonPress = 0;   // Untuk debounce

// Path file counter di SPIFFS
const char* COUNTER_FILE = "/session_counters.dat";

// ============================================================
// FUNGSI: Inisialisasi Kamera OV2640
// ============================================================
bool initCamera() {
  camera_config_t config;
  config.ledc_channel = LEDC_CHANNEL_0;
  config.ledc_timer   = LEDC_TIMER_0;
  config.pin_d0       = Y2_GPIO_NUM;
  config.pin_d1       = Y3_GPIO_NUM;
  config.pin_d2       = Y4_GPIO_NUM;
  config.pin_d3       = Y5_GPIO_NUM;
  config.pin_d4       = Y6_GPIO_NUM;
  config.pin_d5       = Y7_GPIO_NUM;
  config.pin_d6       = Y8_GPIO_NUM;
  config.pin_d7       = Y9_GPIO_NUM;
  config.pin_xclk     = XCLK_GPIO_NUM;
  config.pin_pclk     = PCLK_GPIO_NUM;
  config.pin_vsync    = VSYNC_GPIO_NUM;
  config.pin_href     = HREF_GPIO_NUM;
  config.pin_sccb_sda = SIOD_GPIO_NUM;
  config.pin_sccb_scl = SIOC_GPIO_NUM;
  config.pin_pwdn     = PWDN_GPIO_NUM;
  config.pin_reset    = RESET_GPIO_NUM;
  config.xclk_freq_hz = 20000000;        // 20 MHz XCLK
  config.pixel_format = PIXFORMAT_JPEG;   // Format JPEG
  config.frame_size   = FRAMESIZE_96X96;  // Resolusi 96x96
  config.jpeg_quality = 12;               // Kualitas JPEG (0-63, makin kecil makin bagus)
  config.fb_count     = 1;                // 1 frame buffer
  config.fb_location  = CAMERA_FB_IN_PSRAM;  // Simpan di PSRAM
  config.grab_mode    = CAMERA_GRAB_WHEN_EMPTY;

  // Inisialisasi kamera
  esp_err_t err = esp_camera_init(&config);
  if (err != ESP_OK) {
    Serial.printf("[ERROR] Kamera gagal init! Kode error: 0x%x\n", err);
    return false;
  }

  Serial.println("[OK] Kamera berhasil diinisialisasi.");
  Serial.println("     Resolusi: 96x96, Format: JPEG, Kualitas: 12");
  return true;
}

// ============================================================
// FUNGSI: Inisialisasi SD Card
// ============================================================
bool initSDCard() {
  if (!SD.begin(SD_CS_PIN)) {
    Serial.println("[ERROR] SD card tidak terbaca!");
    Serial.println("        Periksa koneksi SD card dan pin CS.");
    return false;
  }

  uint8_t cardType = SD.cardType();
  if (cardType == CARD_NONE) {
    Serial.println("[ERROR] Tidak ada SD card yang terdeteksi!");
    return false;
  }

  // Tampilkan info SD card
  Serial.print("[OK] SD card terdeteksi. Tipe: ");
  switch (cardType) {
    case CARD_MMC:  Serial.println("MMC");  break;
    case CARD_SD:   Serial.println("SD");   break;
    case CARD_SDHC: Serial.println("SDHC"); break;
    default:        Serial.println("UNKNOWN"); break;
  }

  uint64_t totalBytes = SD.totalBytes();
  uint64_t usedBytes  = SD.usedBytes();
  Serial.printf("     Total: %.1f MB | Terpakai: %.1f MB | Sisa: %.1f MB\n",
                totalBytes / (1024.0 * 1024.0),
                usedBytes / (1024.0 * 1024.0),
                (totalBytes - usedBytes) / (1024.0 * 1024.0));

  return true;
}

// ============================================================
// FUNGSI: Buat folder kelas di SD card jika belum ada
// ============================================================
void createClassFolders() {
  for (int i = 0; i < NUM_CLASSES; i++) {
    String path = "/" + String(CLASS_NAMES[i]);
    if (!SD.exists(path)) {
      if (SD.mkdir(path)) {
        Serial.printf("[OK] Folder '%s' berhasil dibuat.\n", path.c_str());
      } else {
        Serial.printf("[WARNING] Gagal membuat folder '%s'.\n", path.c_str());
      }
    } else {
      Serial.printf("[OK] Folder '%s' sudah ada.\n", path.c_str());
    }
  }
}

// ============================================================
// FUNGSI: Load session counter dari SPIFFS
// Menyimpan counter agar persistent antar reboot
// ============================================================
void loadSessionCounters() {
  if (!SPIFFS.begin(true)) {
    Serial.println("[WARNING] SPIFFS gagal init. Counter dimulai dari 0.");
    return;
  }

  if (SPIFFS.exists(COUNTER_FILE)) {
    File f = SPIFFS.open(COUNTER_FILE, FILE_READ);
    if (f) {
      // Baca 3 integer counter (satu per kelas)
      for (int i = 0; i < NUM_CLASSES; i++) {
        if (f.available() >= sizeof(int)) {
          f.read((uint8_t*)&sessionCounters[i], sizeof(int));
        }
      }
      f.close();
      Serial.println("[OK] Session counter berhasil di-load dari SPIFFS:");
      for (int i = 0; i < NUM_CLASSES; i++) {
        Serial.printf("     %s: sesi_%03d\n", CLASS_NAMES[i], sessionCounters[i]);
      }
    }
  } else {
    Serial.println("[INFO] File counter belum ada. Mulai dari sesi_001.");
  }
}

// ============================================================
// FUNGSI: Simpan session counter ke SPIFFS
// ============================================================
void saveSessionCounters() {
  File f = SPIFFS.open(COUNTER_FILE, FILE_WRITE);
  if (f) {
    for (int i = 0; i < NUM_CLASSES; i++) {
      f.write((uint8_t*)&sessionCounters[i], sizeof(int));
    }
    f.close();
  } else {
    Serial.println("[WARNING] Gagal menyimpan counter ke SPIFFS!");
  }
}

// ============================================================
// FUNGSI: Tampilkan sisa storage SD card
// ============================================================
void printSDFreeSpace() {
  uint64_t totalBytes = SD.totalBytes();
  uint64_t usedBytes  = SD.usedBytes();
  float freeMB = (totalBytes - usedBytes) / (1024.0 * 1024.0);
  Serial.printf("[STORAGE] Sisa SD card: %.1f MB\n", freeMB);
}

// ============================================================
// FUNGSI: Tampilkan kelas aktif saat ini
// ============================================================
void printKelasAktif() {
  Serial.println("========================================");
  Serial.printf("[KELAS AKTIF] >> %s << (sesi berikutnya: sesi_%03d)\n",
                CLASS_NAMES[kelasAktif],
                sessionCounters[kelasAktif] + 1);
  Serial.println("========================================");
}

// ============================================================
// FUNGSI: Capture 1 sesi (5 frame) dan simpan ke SD card
// ============================================================
void captureSession() {
  if (!sdReady) {
    Serial.println("[ERROR] SD card tidak siap! Capture dibatalkan.");
    return;
  }

  if (!cameraReady) {
    Serial.println("[ERROR] Kamera tidak siap! Capture dibatalkan.");
    return;
  }

  // Increment session counter
  sessionCounters[kelasAktif]++;
  int sesiNum = sessionCounters[kelasAktif];

  // Buat path folder sesi: /[kelas]/sesi_[NNN]
  char sesiFolderPath[64];
  snprintf(sesiFolderPath, sizeof(sesiFolderPath), "/%s/sesi_%03d",
           CLASS_NAMES[kelasAktif], sesiNum);

  // Buat folder sesi
  if (!SD.mkdir(sesiFolderPath)) {
    Serial.printf("[ERROR] Gagal membuat folder '%s'!\n", sesiFolderPath);
    sessionCounters[kelasAktif]--;  // Rollback counter
    return;
  }

  Serial.println("----------------------------------------");
  Serial.printf("[CAPTURE] Kelas: %s | Sesi: sesi_%03d\n",
                CLASS_NAMES[kelasAktif], sesiNum);
  Serial.println("----------------------------------------");

  int framesOK = 0;

  // Capture 5 frame berturut-turut
  for (int frameIdx = 1; frameIdx <= FRAMES_PER_SESSION; frameIdx++) {
    Serial.printf("[FRAME] Capturing frame %d/%d ...\n", frameIdx, FRAMES_PER_SESSION);

    // Ambil frame dari kamera
    camera_fb_t* fb = esp_camera_fb_get();

    if (fb == NULL) {
      // Frame null — skip dan lanjut
      Serial.printf("[WARNING] Frame %d/%d NULL! Dilewati.\n", frameIdx, FRAMES_PER_SESSION);
      continue;
    }

    // Buat path file: /[kelas]/sesi_[NNN]/f[1-5].jpg
    char filePath[80];
    snprintf(filePath, sizeof(filePath), "%s/f%d.jpg", sesiFolderPath, frameIdx);

    // Simpan ke SD card
    File file = SD.open(filePath, FILE_WRITE);
    if (file) {
      file.write(fb->buf, fb->len);
      file.close();

      float fileSizeKB = fb->len / 1024.0;
      Serial.printf("[OK] Frame %d/%d tersimpan: %s (%.1f KB)\n",
                    frameIdx, FRAMES_PER_SESSION, filePath, fileSizeKB);
      framesOK++;
    } else {
      Serial.printf("[ERROR] Gagal menulis file '%s'!\n", filePath);
    }

    // Kembalikan frame buffer ke pool
    esp_camera_fb_return(fb);

    // Jeda antar frame (kecuali frame terakhir)
    if (frameIdx < FRAMES_PER_SESSION) {
      delay(FRAME_DELAY_MS);
    }
  }

  Serial.println("----------------------------------------");
  Serial.printf("[SELESAI] Sesi sesi_%03d: %d/%d frame berhasil disimpan.\n",
                sesiNum, framesOK, FRAMES_PER_SESSION);

  // Simpan counter ke SPIFFS
  saveSessionCounters();

  // Tampilkan sisa storage
  printSDFreeSpace();

  Serial.println("----------------------------------------");
  Serial.println("[INFO] Tekan tombol untuk capture sesi berikutnya.");
  printKelasAktif();
}

// ============================================================
// FUNGSI: Cek input Serial untuk ganti kelas
// ============================================================
void checkSerialInput() {
  if (Serial.available() > 0) {
    char input = Serial.read();

    // Flush sisa karakter di buffer (misal newline)
    while (Serial.available() > 0) {
      Serial.read();
    }

    switch (input) {
      case 'k':
      case 'K':
        kelasAktif = 0;
        Serial.println("\n[GANTI KELAS] Kelas aktif diubah ke: KERTAS");
        printKelasAktif();
        break;

      case 'p':
      case 'P':
        kelasAktif = 1;
        Serial.println("\n[GANTI KELAS] Kelas aktif diubah ke: PLASTIK");
        printKelasAktif();
        break;

      case 'o':
      case 'O':
        kelasAktif = 2;
        Serial.println("\n[GANTI KELAS] Kelas aktif diubah ke: ORGANIK");
        printKelasAktif();
        break;

      default:
        Serial.printf("[INFO] Karakter '%c' tidak dikenali.\n", input);
        Serial.println("       Gunakan: 'k'=kertas, 'p'=plastik, 'o'=organik");
        break;
    }
  }
}

// ============================================================
// FUNGSI: Blink LED onboard (untuk indikasi error)
// ============================================================
void blinkLED(int pin, int times, int delayMs) {
  for (int i = 0; i < times; i++) {
    digitalWrite(pin, LOW);   // LED XIAO aktif LOW
    delay(delayMs);
    digitalWrite(pin, HIGH);
    delay(delayMs);
  }
}

// ============================================================
// SETUP
// ============================================================
void setup() {
  // Inisialisasi Serial
  Serial.begin(115200);
  delay(2000);  // Tunggu Serial siap

  Serial.println("============================================");
  Serial.println("  XIAO ESP32S3 Sense — Capture Dataset");
  Serial.println("  Klasifikasi Sampah: CNN + LSTM");
  Serial.println("============================================");
  Serial.println();

  // Konfigurasi pin
  pinMode(BUTTON_PIN, INPUT_PULLUP);
  pinMode(LED_BUILTIN_PIN, OUTPUT);
  digitalWrite(LED_BUILTIN_PIN, HIGH);  // LED mati (aktif LOW)

  // --- Inisialisasi Kamera ---
  Serial.println("[INIT] Menginisialisasi kamera OV2640...");
  cameraReady = initCamera();

  if (!cameraReady) {
    Serial.println("[FATAL] Kamera gagal! Sistem berhenti.");
    Serial.println("        Periksa koneksi kamera dan reboot.");
    // Loop blink LED onboard sebagai indikator error
    while (true) {
      blinkLED(LED_BUILTIN_PIN, 3, 200);
      delay(1000);
    }
  }

  // --- Inisialisasi SD Card ---
  Serial.println("\n[INIT] Menginisialisasi SD card...");
  sdReady = initSDCard();

  if (sdReady) {
    // Buat folder kelas
    Serial.println("\n[INIT] Menyiapkan folder kelas...");
    createClassFolders();
  } else {
    Serial.println("[WARNING] SD card tidak tersedia.");
    Serial.println("         Masukkan SD card dan reboot.");
  }

  // --- Load Session Counter dari SPIFFS ---
  Serial.println("\n[INIT] Memuat session counter dari SPIFFS...");
  loadSessionCounters();

  // --- Tampilkan status awal ---
  Serial.println("\n============================================");
  Serial.println("  SISTEM SIAP — Menunggu Perintah");
  Serial.println("============================================");
  Serial.println("  Tombol : tekan untuk capture 5 frame");
  Serial.println("  Serial : 'k'=kertas 'p'=plastik 'o'=organik");
  Serial.println("============================================\n");
  printKelasAktif();
}

// ============================================================
// LOOP UTAMA
// ============================================================
void loop() {
  // Cek input Serial untuk ganti kelas
  checkSerialInput();

  // Cek tombol capture (aktif LOW karena INPUT_PULLUP)
  if (digitalRead(BUTTON_PIN) == LOW) {
    // Debounce sederhana
    unsigned long now = millis();
    if (now - lastButtonPress > DEBOUNCE_DELAY_MS) {
      lastButtonPress = now;

      // Nyalakan LED saat capture
      digitalWrite(LED_BUILTIN_PIN, LOW);

      // Lakukan capture sesi
      captureSession();

      // Matikan LED
      digitalWrite(LED_BUILTIN_PIN, HIGH);
    }
  }

  // Delay kecil untuk mengurangi polling
  delay(10);
}
